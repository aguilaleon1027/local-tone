2.실행할 SQL

  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  title TEXT NOT NULL,
  category TEXT,
  color TEXT,
  size TEXT,
  image_url TEXT NOT NULL,
  thumbnail_url TEXT,
  is_available BOOLEAN DEFAULT true,
  created_at TIMESTAMP DEFAULT now()
